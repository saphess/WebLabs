Удаление всех HTML тегов $pattern = "/<[^>]+>/";
Поиск всех тегов <b> (жирный текст) $pattern = "/<b>(.*?)<\/b>/";
Поиск всех ссылок $pattern = "/<a\s+href=['\"](.*?)['\"][^>]*>(.*?)<\/a>/";
Поиск всех тегов <p></p> $pattern = "/<p>(.*?)<\/p>/";
Поиск всех тегов <h1></h1> $pattern = "/<h1>(.*?)<\/h1>/";
